<script>
  (function() {
    'use strict';
    
    function moveSearchToSidebar() {
      // Navbar'daki search button'u bul
      const navbarSearch = document.querySelector('header button[aria-label*="earch" i], header [class*="search"]');
      const sidebar = document.querySelector('aside, nav[class*="sidebar"]');
      
      if (!navbarSearch || !sidebar) {
        return;
      }
      
      // Eğer sidebar'da zaten custom search varsa, tekrar ekleme
      if (sidebar.querySelector('.apinizer-custom-search')) {
        return;
      }
      
      // Navbar search'ü gizle
      navbarSearch.style.display = 'none';
      
      // Yeni search butonu oluştur
      const customSearch = document.createElement('button');
      customSearch.className = 'apinizer-custom-search';
      customSearch.innerHTML = navbarSearch.innerHTML;
      
      // Stil ver
      customSearch.style.cssText = `
        display: flex !important;
        align-items: center;
        gap: 8px;
        width: calc(100% - 24px);
        margin: 12px;
        padding: 10px 12px;
        background: rgba(0, 0, 0, 0.05);
        border: 1px solid rgba(0, 0, 0, 0.1);
        border-radius: 8px;
        font-size: 14px;
        cursor: pointer;
        transition: all 0.2s;
        text-align: left;
      `;
      
      // Click event ekle
      customSearch.addEventListener('click', function(e) {
        e.preventDefault();
        e.stopPropagation();
        navbarSearch.click();
      });
      
      // Hover effect
      customSearch.addEventListener('mouseenter', function() {
        this.style.background = 'rgba(0, 0, 0, 0.08)';
      });
      
      customSearch.addEventListener('mouseleave', function() {
        this.style.background = 'rgba(0, 0, 0, 0.05)';
      });
      
      // Sidebar'ın en başına ekle
      sidebar.insertBefore(customSearch, sidebar.firstElementChild);
    }
    
    // Sayfa yüklendiğinde çalıştır
    function init() {
      moveSearchToSidebar();
      
      // Dinamik değişiklikleri izle
      const observer = new MutationObserver(function() {
        moveSearchToSidebar();
      });
      
      observer.observe(document.body, {
        childList: true,
        subtree: true
      });
    }
    
    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', init);
    } else {
      init();
    }
    
    // Fallback - her 1 saniyede kontrol et
    setInterval(moveSearchToSidebar, 1000);
  })();
</script>

<style>
  /* Navbar'daki search'ü tamamen gizle */
  header button[aria-label*="earch" i],
  header button[aria-label*="Search" i],
  header [class*="search"] {
    display: none !important;
    visibility: hidden !important;
  }
  
  /* Dark mode desteği */
  @media (prefers-color-scheme: dark) {
    .apinizer-custom-search {
      background: rgba(255, 255, 255, 0.05) !important;
      border-color: rgba(255, 255, 255, 0.1) !important;
      color: rgba(255, 255, 255, 0.9) !important;
    }
    
    .apinizer-custom-search:hover {
      background: rgba(255, 255, 255, 0.08) !important;
    }
  }
  
  /* Search icon boyutu */
  .apinizer-custom-search svg {
    width: 16px;
    height: 16px;
    opacity: 0.6;
  }
</style>
