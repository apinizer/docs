---
title: Sending WSA Setting with Specific Value
description: "You can add WS-Addressing (WSA) settings with specific values to SOAP header using XSLT. Can be used for services that need to be sent with WS-Addressing true, specific action and default wsa:To selected in SoapUI."
---

```xml
<xsl:stylesheet version="1.0" xmlns:xsl="http://www.w3.org/1999/XSL/Transform" xmlns:soap="http://schemas.xmlsoap.org/soap/envelope/" xmlns:tem="http://tempuri.org/" xmlns:wsa="http://www.w3.org/2005/08/addressing">
  <xsl:output method="xml" version="1.0" encoding="UTF-8" indent="yes" />
  <xsl:strip-space elements="*" />
  <xsl:template match="@*|node()">
    <xsl:copy>
      <xsl:apply-templates select="@*|node()" />
    </xsl:copy>
  </xsl:template>
  <xsl:template match="/soap:Envelope">
    <xsl:copy>
      <soap:Header>
        <wsa:Action>
          #Desired Action to Send#
        </wsa:Action>
        <wsa:To>
          #"To" address written in wsdl#
        </wsa:To>
      </soap:Header>
      <xsl:apply-templates />
    </xsl:copy>
  </xsl:template>
  <xsl:template match="soap:Header" />
</xsl:stylesheet>
```
