---
title: Adding Information to SOAP Header
description: "You can add custom information to SOAP header using XSLT. You can add custom fields such as username to message header."
---

```xml
<xsl:stylesheet version="1.0" xmlns:xsl="http://www.w3.org/1999/XSL/Transform" xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/">
  <xsl:output method="xml" version="1.0" encoding="UTF-8" indent="yes" />
  <xsl:strip-space elements="*" />
  <xsl:template match="@*|node()">
    <xsl:copy>
      <xsl:apply-templates select="@*|node()" />
    </xsl:copy>
  </xsl:template>
  <xsl:template match="/soapenv:Envelope">
    <xsl:copy>
      <soapenv:Header>
        <KullaniciAdi xmlns="apinizer.com">Test123</KullaniciAdi>
      </soapenv:Header>
      <xsl:apply-templates />
    </xsl:copy>
  </xsl:template>
  <xsl:template match="soapenv:Header" />
</xsl:stylesheet>
```
